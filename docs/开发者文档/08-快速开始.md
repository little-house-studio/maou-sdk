# 快速开始

三个可复制运行的示例，从简单到完整。

---

## 示例一：最小对话应用（仅 LLM 层）

独立使用 `ChatSession`，不依赖任何 Agent 系统。

```ts
import { ChatSession } from 'maou-agent/core/llm'

// 配置一个 DeepSeek 预设
const preset = {
  name: 'deepseek',
  protocol: 'openai' as const,
  url: 'https://api.deepseek.com/v1',
  model: 'deepseek-chat',
  key: process.env.DEEPSEEK_API_KEY,
}

const chat = new ChatSession({ preset })

// 非流式对话
const resp = await chat.send('解释量子计算的基本原理')
console.log('回答:', resp.content)
console.log('Token 用量:', resp.usage)

// 流式对话
process.stdout.write('AI: ')
for await (const delta of chat.sendStream('用简单的比喻解释递归')) {
  process.stdout.write(delta.delta ?? '')
}
console.log()

// 中断流式输出
setTimeout(() => chat.abort(), 3000)
```

## 示例二：完整 Agent 运行时（Harness 层）

使用 MaouHarness 运行完整的 Agent，处理工具调用。

```ts
import { MaouHarness } from 'maou-agent/harness'

const harness = new MaouHarness({
  projectRoot: '/path/to/your-project',
  agentRoundLimit: 10,  // 最多 10 轮
})

// 运行 Agent
const stream = harness.runAgent({
  sessionId: 'demo-session',
  userMessage: '读取 README.md 的前 10 行，然后总结一下项目是做什么的',
})

for await (const event of stream) {
  switch (event.type) {
    case 'assistant_delta':
      // 实时流式输出
      process.stdout.write(event.data.delta as string)
      break
    case 'tool_call':
      console.log('\n[调用工具]', event.data.toolName, event.data.parameters)
      break
    case 'tool_result':
      console.log('[工具返回]', (event.data.result as any)?.message)
      break
    case 'done':
      console.log('\n[完成]', '共使用了', event.data.usage, 'tokens')
      break
  }
}
```

## 示例三：注册并使用自定义工具

```ts
import {
  Tool,
  ToolRegistry,
  ToolExecutor,
  registerBuiltins,
  createToolResponse,
} from 'maou-agent/core/tools'
import type { ToolDefinition, ToolContext, ToolResponse } from 'maou-agent/core/tools'

// 1. 定义自定义工具
class WebSearchTool extends Tool {
  readonly definition: ToolDefinition = {
    name: 'web_search',
    aliases: ['search'],
    description: '搜索网络内容',
    parameters: {
      type: 'object',
      properties: {
        query: { type: 'string', description: '搜索关键词' },
        max_results: { type: 'integer', description: '最大结果数' },
      },
      required: ['query'],
    },
    allowedModes: ['execute'],
  }

  async execute(
    params: Record<string, unknown>,
    ctx: ToolContext,
  ): Promise<ToolResponse> {
    const query = params.query as string
    const max = (params.max_results as number) ?? 5

    // 模拟搜索
    const results = [{ title: '结果 1', url: 'https://...' }]

    return createToolResponse(true, `搜索 "${query}" 完成`, {
      payload: { query, results, count: results.length },
    })
  }
}

// 2. 注册
const registry = new ToolRegistry()
registerBuiltins(registry)                    // 内置工具
registry.register(new WebSearchTool())        // 自定义工具

// 3. 执行
const executor = new ToolExecutor(registry, {
  defaultTimeoutMs: 15_000,
})

const ctx: ToolContext = {
  sessionId: 'test',
  projectRoot: '/tmp',
  promptRoot: 'ROLE/default',
  sandboxRoot: '/tmp/sandbox',
  sandboxMode: 'normal',
  agentName: 'test-agent',
  agentMode: 'execute',
  pluginSettings: {},
  workingDir: '/tmp',
}

const result = await executor.executeSingle(
  { id: 'call_1', name: 'web_search', parameters: { query: 'maou agent sdk' } },
  ctx,
)

console.log(result.result.ok, result.result.message)
console.log(result.result.payload)
```
